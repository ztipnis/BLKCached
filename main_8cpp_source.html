98 inline void handle_command(int fd){
99  static auto &s = BLKCACHE::Store<4096>::getInst();
100  std::string cmd = std::to_string('\0');
101  cmd = srecv(fd);
102  std::stringstream cmds(cmd);
103  std::string b;
104  if(cmds >> b){
105  if(boost::iequals(b,"set") || boost::iequals(b,"replace")){
106  std::string k,_,e;
107  cmds >> k >> _ >> e >> _;
108  if(k != "")
109  {
110  std::string v = srecv(fd);
111  if(v != ""){
112  (*s).put(k, boost::trim_copy(v));
113  ssend(fd, "STORED\r\n");
114  }
115  }
116  }else if(boost::iequals(b,"get")){
117  std::string k;
118  cmds >> k;
119  if(k != "" && (*s).containsKey(k))
120  ssend(fd, (*s).get(k) + "\r\n");
121  else
122  ssend(fd, "ERROR\r\n");
123  }else if(boost::iequals(b,"delete")){
124  std::string k;
125  cmds >> k;
126  if(k != "" && (*s).containsKey(k))
127  ssend(fd, (*s).del(k) + "\r\n");
128  else
129  ssend(fd, "ERROR\r\n");
130  }else if(boost::iequals(b,"list")){
131  (*s).forKeys([&fd](std::string s){ssend(fd, s + "\r\n");});
132  }else{
133  ssend(fd, "ERROR\r\n");
134  }

199 inline int init( const int maxEvents, int &listen_fd ){
200  int epfd = epoll_create1(EPOLL_CLOEXEC);
201  if(epfd < 0) throw ServerError(errno, std::string(__FILE__) + " - " + std::to_string(__LINE__));
202  int lsfd = socket(AF_INET, SOCK_STREAM | SOCK_NONBLOCK, 0);
203  if (lsfd < 0) throw ServerError(errno, std::string(__FILE__) + " - " + std::to_string(__LINE__));
204  int _true = 1;
205  if(setsockopt(lsfd, SOL_SOCKET, SO_REUSEADDR, &_true, sizeof(_true)) < 0) throw ServerError(errno, std::string(__FILE__) + " - " + std::to_string(__LINE__));
206  const unsigned short port = stoi(config["port"]);
207  sockaddr_in addr;
208  addr.sin_family = AF_INET;
209  addr.sin_port = htons(port);
210  addr.sin_addr.s_addr = INADDR_ANY;
211  if(bind(lsfd, reinterpret_cast<struct sockaddr*>(&addr), sizeof(sockaddr_in)) < 0){
212  close(epfd);
213  close(lsfd);
214  throw ServerError(errno, std::string(__FILE__) + " - " + std::to_string(__LINE__));
215  }
216  epoll_event event;
217  event.events = EPOLLIN | EPOLLET;
218  event.data.fd = lsfd;
219  if(epoll_ctl(epfd, EPOLL_CTL_ADD, lsfd, &event)){
220  close(epfd);
221  close(lsfd);
222  throw ServerError(errno, std::string(__FILE__) + " - " + std::to_string(__LINE__));
223  }
